function bubbleSort(arr) {
    var len = arr.length,
        j, tmp;

    while (len--) {
        for (j = len - 1; j >= 0; j--) {
            if (arr[j + 1] > arr[j]) {
                tmp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = tmp;
            }
        }
    }

    return arr;
}
