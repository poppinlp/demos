function SmallTable() {
            this.desc = function () {
                console.log('small table');
            }
        };

        function LargeTable() {
            this.desc = function () {
                console.log('large table');
            }
        };

        var Factory = {
            createTable: function(type) {
                switch(type) {
                    case 'small':
                        return new SmallTable();
                        break;
                    case 'large':
                        return new LargeTable();
                        break;
                }
            }
        };

        var smallTable = Factory.createTable('small');
        smallTable.desc();
